/**
 * Файл: client.cpp
 * 
 * Консольный клиент мессенджера
 * 
 * Как работает клиент:
 * 1. Подключается к серверу по TCP на порт 8080
 * 2. Запускает два потока:
 *    - Главный поток: читает ввод пользователя и отправляет на сервер
 *    - Поток receiver: асинхронно принимает сообщения от сервера
 * 3. Отображает историю чата и новые сообщения в реальном времени
 * 
 * Многопоточность (выполнение требования задания):
 * - Один поток для отправки сообщений (главный)
 * - Один поток для приёма сообщений (receive_messages)
 * 
 * Протокол: текстовые сообщения, разделённые символом новой строки (\n)
 * 
 * Поддержка ANSI escape-кодов для:
 * - Очистки экрана (\033[2J\033[H)
 * - Удаления строки ввода (\033[1A\033[2K)
 */

#include <iostream>      // Для cout, cin, cerr
#include <string>        // Для std::string
#include <cstring>       // Для memset()
#include <thread>        // Для std::thread (многопоточность)
#include <sys/socket.h>  // Для socket(), connect(), send(), recv()
#include <netinet/in.h>  // Для struct sockaddr_in, htons()
#include <arpa/inet.h>   // Для inet_pton()
#include <unistd.h>      // Для close()

using namespace std;

// ============================================================================
// ГЛОБАЛЬНЫЕ ПЕРЕМЕННЫЕ
// ============================================================================

/**
 * Дескриптор сокета для соединения с сервером
 * 
 * Глобальная переменная, так как используется и в главном потоке,
 * и в потоке приёма сообщений. При завершении работы закрывается.
 */
int sock;

/**
 * Флаг работы клиента
 * 
 * true - клиент работает, false - пора завершать все потоки
 * Используется для корректного завершения обоих потоков
 */
bool running = true;

/**
 * Флаг нахождения внутри чата
 * 
 * true - пользователь в личном или групповом чате
 * false - пользователь в главном меню
 * 
 * Нужен для правильного форматирования вывода:
 * - Если в чате, после отправки сообщения стираем строку ввода
 * - Если в меню, оставляем как есть
 */
bool in_chat = false;

// ============================================================================
// ПОТОК ПРИЁМА СООБЩЕНИЙ
// ============================================================================

/**
 * Функция для работы потока приёма сообщений от сервера
 * 
 * Этот поток запускается один раз при старте клиента и работает
 * параллельно с главным потоком ввода.
 * 
 * Что делает:
 * 1. Постоянно читает данные из сокета
 * 2. Накопливает неполные сообщения (TCP потоковый протокол)
 * 3. Разбирает сообщения по символу \n
 * 4. Выводит полученные сообщения на экран
 * 
 * Особенности:
 * - Обрабатывает специальную команду __REFRESH_CHAT__ для очистки экрана
 * - Работает с ANSI escape-кодами для красивого отображения
 * - При отключении сервера завершает работу клиента
 * 
 * Важно: этот метод запускается в отдельном потоке, поэтому
 * взаимодействие с глобальными переменными должно быть аккуратным
 */
void receive_messages() {
    // Буфер для приёма данных от сервера
    char buffer[4096];
    
    // Буфер для накопления неполных сообщений
    // TCP может принести сообщение не целиком, поэтому накапливаем
    string incomplete;
    
    // Режим перерисовки чата после команды /clear
    // Когда сервер отправляет __REFRESH_CHAT__, мы очищаем экран
    // и показываем историю заново
    bool refresh_mode = false;
    
    // Основной цикл приёма сообщений
    while (running) {
        // Очищаем буфер и принимаем данные от сервера
        memset(buffer, 0, sizeof(buffer));
        int bytes = recv(sock, buffer, sizeof(buffer) - 1, 0);
        
        if (bytes > 0) {
            // Добавляем новые данные в буфер неполных сообщений
            incomplete += string(buffer);
            
            // Разбираем буфер на отдельные сообщения
            // Сообщения разделяются символом новой строки \n
            size_t pos;
            while ((pos = incomplete.find('\n')) != string::npos) {
                // Извлекаем одно сообщение (без \n)
                string msg = incomplete.substr(0, pos);
                // Удаляем его из буфера
                incomplete = incomplete.substr(pos + 1);
                
                // ОБРАБОТКА СПЕЦИАЛЬНОЙ КОМАНДЫ __REFRESH_CHAT__
                // Эта команда приходит от сервера после /clear
                // Нужно очистить экран и показать историю заново
                if (msg == "__REFRESH_CHAT__") {
                    // ANSI escape-код для очистки экрана и перемещения курсора в начало
                    // \033[2J - очистить весь экран
                    // \033[H - переместить курсор в позицию (1,1)
                    cout << "\033[2J\033[H" << flush;
                    
                    // Печатаем справку по командам (как при входе)
                    cout << "=== Навигация ===" << endl;
                    cout << "[ЧАТЫ]:   /users (онлайн) | /chat <ник> (открыть диалог)" << endl;
                    cout << "[ГРУППЫ]: /mygroups (список) | /gcreate <имя> (создать) | /gjoin <имя> (войти)" << endl;
                    cout << "          /gadd (добавить) | /gkick (исключить) | /gmembers (участники)" << endl;
                    cout << "[ОПЦИИ]:  /reply <id> (ответ) | /fwd <id> (пересылка) | /delete <id> (удалить)" << endl;
                    cout << "          /clear (очистить) | /exit (в меню) | /quit (выход)" << endl;
                    cout << "========================================================================" << endl;
                    
                    // Включаем режим перерисовки
                    // Следующие сообщения будут выводиться как часть новой истории
                    refresh_mode = true;
                    continue;
                }
                
                // ВЫВОД ОБЫЧНЫХ СООБЩЕНИЙ
                if (refresh_mode) {
                    // В режиме перерисовки: выводим сообщения, пока не встретим пустую строку
                    if (msg.empty()) {
                        // Пустая строка - сигнал окончания истории
                        refresh_mode = false;
                        cout << "" << flush;
                    } else {
                        // Выводим сообщение истории
                        cout << msg << endl;
                    }
                } else {
                    // Обычный режим: просто выводим сообщение
                    if (!msg.empty()) {
                        cout << msg << endl;
                    }
                    // Принудительный вывод (не дожидаемся \n)
                    cout << "" << flush;
                }
            }
        } 
        else if (bytes == 0) {
            // Сервер закрыл соединение (bytes == 0 означает graceful shutdown)
            cout << "\n[Сервер отключился]" << endl;
            running = false;  // Сигнал главному потоку завершиться
            break;
        }
        // bytes < 0: ошибка сокета, игнорируем и продолжаем
    }
}

// ============================================================================
// ТОЧКА ВХОДА (main)
// ============================================================================

/**
 * Главная функция клиента
 * 
 * Последовательность действий:
 * 1. Создание TCP-сокета
 * 2. Подключение к серверу на 127.0.0.1:8080
 * 3. Запуск потока для приёма сообщений (receive_messages)
 * 4. Основной цикл: читаем ввод пользователя и отправляем на сервер
 * 5. При команде /quit или ошибке - завершаем работу
 * 
 * Команды, которые понимает клиент:
 * - /chat <ник> - начать личный диалог
 * - /gjoin <группа> - войти в группу
 * - /gcreate <группа> - создать группу
 * - /mygroups - показать мои группы
 * - /users - показать онлайн пользователей
 * - /reply <id> <текст> - ответить на сообщение
 * - /fwd <id> chat/group <куда> - переслать сообщение
 * - /delete <id> - удалить сообщение
 * - /clear - очистить историю чата
 * - /exit - выйти из текущего чата в главное меню
 * - /quit - выйти из программы
 * 
 * Формат отправляемых сообщений: текст + '\n'
 * Сервер ждёт символ новой строки как разделитель сообщений
 */
int main() {
    // ШАГ 1: СОЗДАНИЕ СОКЕТА
    // socket(AF_INET, SOCK_STREAM, 0)
    // - AF_INET: используем IPv4
    // - SOCK_STREAM: используем TCP (надёжная доставка)
    // - 0: автоматический выбор протокола для SOCK_STREAM (TCP)
    sock = socket(AF_INET, SOCK_STREAM, 0);
    if (sock < 0) {
        cerr << "Ошибка сокета" << endl;
        return 1;
    }
    
    // ШАГ 2: НАСТРОЙКА АДРЕСА СЕРВЕРА
    struct sockaddr_in address;
    address.sin_family = AF_INET;              // IPv4
    address.sin_port = htons(8080);            // Порт 8080 (переводим в сетевой порядок)
    // inet_pton: преобразует IP-адрес из текста в бинарный вид
    // 127.0.0.1 - localhost (тот же компьютер)
    inet_pton(AF_INET, "127.0.0.1", &address.sin_addr);
    
    // ШАГ 3: ПОДКЛЮЧЕНИЕ К СЕРВЕРУ
    if (connect(sock, (struct sockaddr*)&address, sizeof(address)) < 0) {
        cerr << "Ошибка подключения. Запустите ./server" << endl;
        return 1;
    }
    
    // ШАГ 4: ЗАПУСК ПОТОКА ДЛЯ ПРИЁМА СООБЩЕНИЙ
    // 
    // ЗАЧЕМ НУЖЕН ВТОРОЙ ПОТОК?
    // 
    // Если бы мы принимали сообщения в главном потоке, то не могли бы
    // одновременно читать ввод пользователя - программа бы зависала.
    // 
    // Два потока работают параллельно:
    // - Главный поток: ждёт, пока пользователь введёт сообщение
    // - Поток receiver: ждёт сообщения от сервера
    // 
    // Это и есть МНОГОПОТОЧНОСТЬ, требуемая заданием:
    // "у клиента будет два потока, один отправляет сообщения на сервер,
    //  а второй их получает"
    thread receiver(receive_messages);
    
    // ШАГ 5: ОСНОВНОЙ ЦИКЛ ОТПРАВКИ СООБЩЕНИЙ
    string input;  // Строка для ввода пользователя
    
    while (running) {
        // Читаем строку из стандартного ввода (ждём, пока пользователь нажмёт Enter)
        getline(cin, input);
        
        // КОМАНДА ВЫХОДА /quit
        if (input == "/quit") {
            // Отправляем команду на сервер (с \n в конце)
            send(sock, (input + "\n").c_str(), input.length() + 1, 0);
            // Сигналим потоку приёма, что пора завершаться
            running = false;
            break;
        }
        
        // ОТСЛЕЖИВАНИЕ СОСТОЯНИЯ (внутри чата или в меню)
        // Нужно для правильного форматирования вывода:
        // - Если мы внутри чата, после отправки сообщения стираем строку ввода
        // - Сообщение не должно дублироваться на экране
        // 
        // Команды, которые переключают в режим чата:
        if (input.substr(0, 6) == "/chat " || input.substr(0, 7) == "/gjoin ") {
            in_chat = true;   // Пользователь зашёл в личный или групповой чат
        }
        
        // Команды, которые выходят из режима чата:
        if (input == "/exit" || input == "/gleave") {
            in_chat = false;  // Пользователь вернулся в главное меню
        }
        
        // ОТПРАВКА СООБЩЕНИЯ НА СЕРВЕР
        // Добавляем \n в конце, как требует протокол
        // +1 для символа \0 (строка C-style)
        send(sock, (input + "\n").c_str(), input.length() + 1, 0);
        
        // КОРРЕКТИРОВКА ОТОБРАЖЕНИЯ (стираем строку ввода)
        // Если мы в чате и отправили обычное сообщение (не команду),
        // стираем строку ввода, чтобы сообщение не дублировалось на экране.
        // 
        // Пример без стирания:
        //   [you]: Привет!      <- пользователь ввёл
        //   [you]: Привет!      <- эхо от сервера (дублирование)
        // 
        // Со стиранием:
        //   [you]: Привет!      <- только эхо от сервера
        // 
        // Условия:
        // 1. Пользователь в чате (in_chat == true)
        // 2. Сообщение не начинается с '/' (иначе это команда, не нужно стирать)
        // 3. Сообщение не является командой /chat или /gjoin
        // 
        // ANSI escape-код:
        // \033[1A - переместить курсор на одну строку вверх
        // \033[2K - очистить текущую строку
        if (in_chat && input[0] != '/' && 
            input.substr(0, 6) != "/chat " && 
            input.substr(0, 6) != "/gjoin") {
            cout << "\033[1A\033[2K" << flush;
        }
    }
    
    // ШАГ 6: ЗАВЕРШЕНИЕ РАБОТЫ
    // Закрываем сокет (сигналит потоку receiver об ошибке чтения)
    close(sock);
    
    // Ждём завершения потока приёма сообщений
    // join() блокирует выполнение, пока поток не завершится
    receiver.join();
    
    return 0;
}
